expression     → equality ;
equality       → comparison ( ( "!=" | "==" ) comparison )* ;
comparison     → addition ( ( ">" | ">=" | "<" | "<=" ) addition )* ;
addition       → multiplication ( ( "-" | "+" ) multiplication )* ;
multiplication → unary ( ( "/" | "*" ) unary )* ;
unary          → "!" ("!")* boolean
               | "-" ("-")* NUMBER
               | primary ;
boolean        → "false" | "true"
primary        → NUMBER | STRING | boolean | "nil"
               | "(" expression ")" ;
